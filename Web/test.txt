int binarySearch(arr[], low, high, val){
    if(low more than high){
        return -1
    }

    mid = low + (high-low)/2

    if(arr[mid] == val){
        return mid;
    } else if(arr[mid]> val){
        return binarySearch(arr[], low, mid-1, val)
    }else if(arr[mid]>val){
        return binarySearch(arr[], mid +1, high, val)
    }
}
